Chapter#06 Math expressions: Unfamiliar operators

  • var num1 = 5;
    Result: num1++ = 5;
    Result: num1 = 6;

  • var numb = 20;
    Result: numb = 20;
    numb = numb++;
    Result: numb = 21;

  • var number1 = 15;
    Result: number1-- = 15;
    Result: number1 = 14;

  • var number2 = 10;
    Result: number2 = 10;
    number2 = number2--;
    Result: number2 = 9;

  • var num = 25;
    Result: ++num = 26;
    Result: num = 26;

  • var num2 = 30;
    Result: --num2 = 29;
    Result: num2 = 29;
