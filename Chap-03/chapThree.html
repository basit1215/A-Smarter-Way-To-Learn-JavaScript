Chapter#03 Variables of Numbers

  • var weight= 40;
    weight + 20;
    Result: weight = 40;

  • var orgNum = 25;
    var updtNum = orgNum + 10;
    Result: updtNum = 35;

  • var numb = 25;
    var numbToAdd = 10;
    Result: numb + numbToAdd = 35;

  • var num = 25;
    num = num + 10;
    Result: num = 35;

  • var originalNum = "25";
    var newNum = originalNum + 10;
    Result: newNum = "2510";
